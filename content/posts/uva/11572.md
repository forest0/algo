// UVA 11572
#include <cstdio>
#include <set>
#include <algorithm>
using std::set;
using std::max;

const int maxn = 1000000 + 5;
int a[maxn];
int n;

int solve() {
    int L = 0, R = 0, ans = 0;
    set<int> s;
    while (R < n) {
        while (R < n && !s.count(a[R])) s.insert(a[R++]);
        ans = max(ans, R-L);
        s.erase(a[L++]);
    }
    return ans;
}

int main(int argc, char *argv[]) {
    int T; scanf("%d", &T);
    while (T--) {
        scanf("%d", &n);
        for (int i = 0; i < n; ++i) {
            scanf("%d", &a[i]);
        }
        printf("%d\n", solve());
    }
    return 0;
}
