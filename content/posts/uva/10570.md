## 分析

穷举起点1的位置，然后依次将每个数放到正确位置即可。

## 实现

### 实现1

1300ms AC, 注意最内层循环可以用一个字典优化掉

```cpp
// UVa 10570
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int maxn = 500+5;
const int dr[] = {1, -1};

int n, a[maxn], b[maxn];

int main(int argc, char *argv[]) {
    // freopen("input.txt", "r", stdin);
    while (scanf("%d", &n) == 1 && n) {
        for (int i = 0; i < n; ++i) scanf("%d", &a[i]);
        memcpy(b, a, n*sizeof(int)); // 备份

        int ans = 0x3f3f3f3f;
        for (int i = 0; i < n; ++i) { // 起点
            for (auto d : dr) {
                int cnt = 0;
                memcpy(a, b, n*sizeof(int));
                for (int j = 0; j < n; ++j) {
                    int idx = (i+j*d) % n; if (idx < 0) idx += n;
                    if (a[idx] == (j+1)) continue;
                    for (int k = 0; k < n; ++k) if (a[k] == (j+1)) {
                        swap(a[k], a[idx]);
                        ++cnt;
                        break;
                    }
                    if (cnt >= ans) break;
                }
                ans = min(ans, cnt);
            }
        }
        printf("%d\n", ans);
    }
    return 0;
}
```

### 实现2

60ms AC

```cpp
// UVa 10570
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int maxn = 500+5;
const int dr[] = {1, -1};

int n, a[maxn], b[maxn], pos[maxn], posB[maxn];

int main(int argc, char *argv[]) {
    // freopen("input.txt", "r", stdin);
    while (scanf("%d", &n) == 1 && n) {
        for (int i = 0; i < n; ++i) {
            scanf("%d", &b[i]);
            posB[b[i]] = i;
        }

        int ans = 0x3f3f3f3f;
        for (int i = 0; i < n; ++i) { // 起点
            for (auto d : dr) {
                int cnt = 0;
                memcpy(a, b, n*sizeof(int));
                memcpy(pos, posB, n*sizeof(int));
                for (int j = 0; j < n; ++j) {
                    int idx = (i+j*d) % n; if (idx < 0) idx += n;
                    if (a[idx] == (j+1)) continue;
                    pos[a[idx]] = pos[j+1];
                    swap(a[pos[j+1]], a[idx]);
                    pos[j+1] = idx;
                    if (++cnt >= ans) break;
                }
                ans = min(ans, cnt);
            }
        }
        printf("%d\n", ans);
    }
    return 0;
}
```
