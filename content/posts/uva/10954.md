// UVA 10954
#include <cstdio>
#include <queue>
using std::priority_queue;


int main(int argc, char *argv[]) {
    int n, d;
    while (scanf("%d", &n) == 1 && n) {
        priority_queue<int, std::vector<int>, std::greater<int> > q;
        for (int i = 0; i < n; ++i) {
            scanf("%d", &d);
            q.push(d);
        }
        long long ans = 0;
        for (int i = 0; i < n-1; ++i) {
            long long m = q.top(); q.pop();
            long long n = q.top(); q.pop();
            ans += m+n;
            q.push(m+n);
        }
        printf("%lld\n", ans);
    }
    return 0;
}
